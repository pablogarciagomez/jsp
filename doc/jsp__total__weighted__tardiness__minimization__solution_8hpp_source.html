71  void UpdateHeadsAndTails() const
72  {
73  if (!changes.empty()) {
74  std::vector<std::reference_wrapper<const TaskType>> tasks;
75  GetTasksTopologicalOrder(std::back_inserter(tasks));
76  // update the heads
77  // find the first task whose head has to be updated
78  auto it = tasks.begin();
79  while (it != tasks.end()) {
80  if (changes.count(*it) != 0) {
81  break;
82  }
83  ++it;
84  }
85  // calculate the head of the tasks that are scheduled after the first modified task
86  while (it != tasks.end()) {
87  auto [job_predecessor, machine_predecessor] = GetPrevTasks(*it);
88  disjunctive_graph[*it].head = TimeType{};
89  if (job_predecessor.has_value()) {
90  disjunctive_graph[*it].head =
91  std::max(disjunctive_graph.at(*it).head, disjunctive_graph.at(*job_predecessor).head + job_predecessor->get().GetDuration());
92  }
93  if (machine_predecessor.has_value()) {
94  disjunctive_graph[*it].head = std::max(
95  disjunctive_graph.at(*it).head, disjunctive_graph.at(*machine_predecessor).head + machine_predecessor->get().GetDuration());
96  }
97  ++it;
98  }
99  // update the tails
100  if constexpr (Tails::value) {
101  // find "the last" task whose tails has to be updated
102  auto rit = tasks.rbegin();
103  while (rit != tasks.rend()) {
104  if (changes.count(*rit) != 0) {
105  break;
106  }
107  ++rit;
108  }
109  // calculate the tails of the tasks that are scheduled before "the last" modified task
110  std::vector<std::reference_wrapper<const JobType>> jobs;
111  GetProblem().GetJobs(std::back_inserter(jobs));
112  while (rit != tasks.rend()) {
113  auto [job_successor, machine_successor] = GetNextTasks(*rit);
114 
115  for (const JobType& job: jobs) {
116  disjunctive_graph[*rit].tails[job] = TimeType{};
117  if (job_successor.has_value()) {
118  disjunctive_graph[*rit].tails[job] =
119  std::max(disjunctive_graph.at(*rit).tails.at(job),
120  disjunctive_graph.at(*job_successor).tails.at(job) + job_successor->get().GetDuration());
121  }
122  if (machine_successor.has_value()) {
123  disjunctive_graph[*rit].tails[job] =
124  std::max(disjunctive_graph.at(*rit).tails.at(job),
125  disjunctive_graph.at(*machine_successor).tails.at(job) + machine_successor->get().GetDuration());
126  }
127  }
128  ++rit;
129  }
130  }
131  changes.clear();
132  // calculate the total weighted tardiness
133  std::vector<std::reference_wrapper<const TaskType>> final_tasks;
134  GetProblem().GetFinalTasks(std::back_inserter(final_tasks));
135  total_weighted_tardiness = TimeType{};
136  for (const TaskType& task: final_tasks) {
137  const JobType& job = task.GetJob();
138  auto tardiness = disjunctive_graph.at(task).head + task.GetDuration() - job.GetDueDate();
139  total_weighted_tardiness += std::max(TimeType{}, tardiness) * job.GetWeight();
140  }
141  }

227  const auto recursive_critical_blocks = [this](Iter dest, const auto& lambda, const TaskType& current_task, BlockType current_block) -> Iter {
228  // get previous tasks
229  auto [job_predecessor, machine_predecessor] = GetPrevTasks(current_task);
230 
231  // if there are not any predecessors
232  if (!job_predecessor.has_value() && !machine_predecessor.has_value()) {

234  *dest++ = current_block;
235  }
236  }

238  // if the job predecessor is critical
239  if (job_predecessor.has_value() && EqualTime(job_predecessor->get().GetDuration() + disjunctive_graph.at(*job_predecessor).head,
240  disjunctive_graph.at(current_task).head)) {
241  if (!current_block.Empty()) {
242  *dest++ = current_block;
243  dest = lambda(dest, lambda, *job_predecessor, BlockType{});
244  } else {
245  dest = lambda(dest, lambda, *job_predecessor, current_block);
246  }
247  }
248 
249  // if the machine predecessor is critical
250  if (machine_predecessor.has_value() &&
251  EqualTime(machine_predecessor->get().GetDuration() + disjunctive_graph.at(*machine_predecessor).head,
252  disjunctive_graph.at(current_task).head)) {
253  current_block.AddRestrictionFront(*machine_predecessor, current_task);
254  dest = lambda(dest, lambda, *machine_predecessor, current_block);
255  }
256 
257  return dest;
258  };

260  GetProblem().GetFinalTasks(std::back_inserter(final_tasks));
261  for (const TaskType& final_task: final_tasks) {
262  dest = recursive_critical_blocks(dest, recursive_critical_blocks, final_task, BlockType{});
263  }

423  template <typename Iter> Iter GetTasksTopologicalOrder(Iter dest) const
424  {
425  std::unordered_map<std::reference_wrapper<const TaskType>, unsigned short, std::hash<TaskType>, std::equal_to<TaskType>> in_degree;
426  std::queue<std::reference_wrapper<const TaskType>> available;
427  std::size_t cnt = 0;
428 
429  for (const auto& [task, metadata]: disjunctive_graph) {
430  auto d = (metadata.job_predecessor.has_value() ? 1 : 0) + (metadata.machine_predecessor.has_value() ? 1 : 0);
431  in_degree[task] = d;
432  if (d == 0) {
433  available.push(task);
434  }
435  }
436 
437  while (!available.empty()) {
438  const TaskType& current = available.front();
439  available.pop();
440  *dest++ = current;
441  cnt++;
442  auto [job_successor, machine_successor] = GetNextTasks(current);
443  if (job_successor.has_value()) {
444  if (--in_degree[*job_successor] == 0) {
445  available.push(*job_successor);
446  }
447  }
448  if (machine_successor.has_value()) {
449  if (--in_degree[*machine_successor] == 0) {
450  available.push(*machine_successor);
451  }
452  }
453  }
454 
455  if (cnt != disjunctive_graph.size()) {
456  throw std::invalid_argument("Not a DAG");
457  }
458  return dest;
459  }
