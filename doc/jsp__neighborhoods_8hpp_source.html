40  Iter last,
41  const Solution& solution,
42  const std::optional<std::reference_wrapper<const typename Solution::TaskType>>& before,
43  const std::optional<std::reference_wrapper<const typename Solution::TaskType>>& after)
44 {
45  using TaskType = typename Solution::TaskType;
46  using TimeType = typename Solution::TimeType;
47  //estimate heads
48  std::unordered_map<std::reference_wrapper<const TaskType>, TimeType, std::hash<TaskType>, std::equal_to<TaskType>> heads;
49  std::optional<std::reference_wrapper<const TaskType>> job_predecessor = solution.GetPrevPrecedenceConstrainedTask(*first);
50  heads[*first] = std::max(job_predecessor.has_value() ? solution.GetHead(*job_predecessor) + job_predecessor->get().GetDuration() : TimeType{},
51  before.has_value() ? solution.GetHead(*before) + before->get().GetDuration() : TimeType{});
52  for (auto it = std::next(first); it != last; ++it) {
53  job_predecessor = solution.GetPrevPrecedenceConstrainedTask(*it);
54  heads[*it] = std::max(job_predecessor.has_value() ? solution.GetHead(*job_predecessor) + job_predecessor->get().GetDuration() : TimeType{},
55  heads[*std::prev(it)] + std::prev(it)->get().GetDuration());
56  }
57 
58  //estimate tails
59  std::unordered_map<std::reference_wrapper<const TaskType>, TimeType, std::hash<TaskType>, std::equal_to<TaskType>> tails;
60  std::optional<std::reference_wrapper<const TaskType>> job_successor = solution.GetNextPrecedenceConstrainedTask(*std::prev(last));
61  tails[*std::prev(last)] = std::max(job_successor.has_value() ? solution.GetTail(*job_successor) + job_successor->get().GetDuration() : TimeType{},
62  after.has_value() ? solution.GetTail(*after) + after->get().GetDuration() : TimeType{});
63  for (auto it = std::prev(last); it-- != first;) {
64  job_successor = solution.GetNextPrecedenceConstrainedTask(*it);
65  tails[*it] = std::max(job_successor.has_value() ? solution.GetTail(*job_successor) + job_successor->get().GetDuration() : TimeType{},
66  tails[*std::next(it)] + std::next(it)->get().GetDuration());
67  }
68 
69  //estimate makespan
70  TimeType makespan{};
71  for (auto it = first; it != last; ++it) {
72  makespan = std::max(makespan, heads[*it] + it->get().GetDuration() + tails[*it]);
73  }

154 template <typename Estimate, typename Solution, typename Move, typename Iter>
155 static double GetQuality(Solution& solution,
156  Move& move,
157  [[maybe_unused]] Iter first,
158  [[maybe_unused]] Iter last,
159  const std::optional<std::reference_wrapper<const typename Solution::TaskType>>& before,
160  const std::optional<std::reference_wrapper<const typename Solution::TaskType>>& after)
161 {
162  if constexpr (Estimate::value) {
163  if constexpr (is_specialization<Solution, JSPMakespanMinimizationSolution>::value) {

171  solution.ApplyMove(move.Invert());
172  move.Invert();
173  return quality;
