71  void UpdateHeadsAndTails() const
72  {
73  if (!changes.empty()) {
74  std::vector<std::reference_wrapper<const TaskType>> tasks;
75  GetTasksTopologicalOrder(std::back_inserter(tasks));
76  // update the heads
77  // find the first task whose head has to be updated
78  auto it = tasks.begin();
79  while (it != tasks.end()) {
80  if (changes.count(*it) != 0) {
81  break;
82  }
83  ++it;
84  }
85  // calculate the head of the tasks that are scheduled after the first modified task
86  while (it != tasks.end()) {
87  auto [job_predecessor, machine_predecessor] = GetPrevTasks(*it);
88  disjunctive_graph[*it].head = TimeType{};
89  if (job_predecessor.has_value()) {
90  disjunctive_graph[*it].head =
91  std::max(disjunctive_graph.at(*it).head, disjunctive_graph.at(*job_predecessor).head + job_predecessor->get().GetDuration());
92  }
93  if (machine_predecessor.has_value()) {
94  disjunctive_graph[*it].head = std::max(
95  disjunctive_graph.at(*it).head, disjunctive_graph.at(*machine_predecessor).head + machine_predecessor->get().GetDuration());
96  }
97  ++it;
98  }
99  // update the tails
100  if constexpr (Tails::value) {
101  // find "the last" task whose tail has to be updated
102  auto rit = tasks.rbegin();
103  while (rit != tasks.rend()) {
104  if (changes.count(*rit) != 0) {
105  break;
106  }
107  ++rit;
108  }
109  // calculate the tail of the tasks that are scheduled before "the last" modified task
110  while (rit != tasks.rend()) {
111  auto [job_successor, machine_successor] = GetNextTasks(*rit);
112  disjunctive_graph[*rit].tail = TimeType{};
113  if (job_successor.has_value()) {
114  disjunctive_graph[*rit].tail =
115  std::max(disjunctive_graph.at(*rit).tail, disjunctive_graph.at(*job_successor).tail + job_successor->get().GetDuration());
116  }
117  if (machine_successor.has_value()) {
118  disjunctive_graph[*rit].tail = std::max(
119  disjunctive_graph.at(*rit).tail, disjunctive_graph.at(*machine_successor).tail + machine_successor->get().GetDuration());
120  }
121  ++rit;
122  }
123  }
124  changes.clear();
125  // calculate the latest task to be completed
126  std::vector<std::reference_wrapper<const TaskType>> final_tasks;
127  GetProblem().GetFinalTasks(std::back_inserter(final_tasks));
128  makespan = TimeType{};
129  for (const TaskType& task: final_tasks) {
130  makespan = std::max(makespan, task.GetDuration() + disjunctive_graph.at(task).head);
131  }
132  }

200  template <typename Block, typename Iter> Iter GetCriticalBlocks(Iter dest) const
201  {
202  using BlockType = Block;
203 
204  UpdateHeadsAndTails();

206  const auto recursive_critical_blocks = [this](Iter dest, const auto& lambda, const TaskType& current_task, BlockType current_block) -> Iter {
207  // get previous tasks
208  auto [job_predecessor, machine_predecessor] = GetPrevTasks(current_task);
209 
210  // if there are not any predecessors
211  if (!job_predecessor.has_value() && !machine_predecessor.has_value()) {
212  if (!current_block.Empty()) {
213  *dest++ = current_block;
214  }
215  }
216 
217  // if the job predecessor is critical
218  if (job_predecessor.has_value() && EqualTime(job_predecessor->get().GetDuration() + disjunctive_graph.at(*job_predecessor).head,
219  disjunctive_graph.at(current_task).head)) {
220  if (!current_block.Empty()) {
221  *dest++ = current_block;
222  dest = lambda(dest, lambda, *job_predecessor, BlockType{});
223  } else {
224  dest = lambda(dest, lambda, *job_predecessor, current_block);
225  }
226  }
227 
228  // if the machine predecessor is critical
229  if (machine_predecessor.has_value() &&
230  EqualTime(machine_predecessor->get().GetDuration() + disjunctive_graph.at(*machine_predecessor).head,
231  disjunctive_graph.at(current_task).head)) {
232  current_block.AddRestrictionFront(*machine_predecessor, current_task);
233  dest = lambda(dest, lambda, *machine_predecessor, current_block);
234  }
235 
236  return dest;
237  };
238  std::vector<std::reference_wrapper<const TaskType>> final_tasks;
239  GetProblem().GetFinalTasks(std::back_inserter(final_tasks));
240  for (const TaskType& final_task: final_tasks) {
241  if (EqualTime(makespan, final_task.GetDuration() + disjunctive_graph.at(final_task).head)) {
242  dest = recursive_critical_blocks(dest, recursive_critical_blocks, final_task, BlockType{});
243  }
244  }

404  template <typename Iter> Iter GetTasksTopologicalOrder(Iter dest) const
405  {
406  std::unordered_map<std::reference_wrapper<const TaskType>, unsigned short, std::hash<TaskType>, std::equal_to<TaskType>> in_degree;
407  std::queue<std::reference_wrapper<const TaskType>> available;
408  std::size_t cnt = 0;
409 
410  for (const auto& [task, metadata]: disjunctive_graph) {
411  auto d = (metadata.job_predecessor.has_value() ? 1 : 0) + (metadata.machine_predecessor.has_value() ? 1 : 0);
412  in_degree[task] = d;
413  if (d == 0) {
414  available.push(task);
415  }
416  }
417 
418  while (!available.empty()) {
419  const TaskType& current = available.front();
420  available.pop();
421  *dest++ = current;
422  cnt++;
423  auto [job_successor, machine_successor] = GetNextTasks(current);
424  if (job_successor.has_value()) {
425  if (--in_degree[*job_successor] == 0) {
426  available.push(*job_successor);
427  }
428  }
429  if (machine_successor.has_value()) {
430  if (--in_degree[*machine_successor] == 0) {
431  available.push(*machine_successor);
432  }
433  }
434  }
435 
436  if (cnt != disjunctive_graph.size()) {
437  throw std::invalid_argument("Not a DAG");
438  }
439  return dest;
440  }

576  std::string SolutionSequence() const
577  {
578  std::string solution;

580  GetTasksTopologicalOrder(std::back_inserter(ordered_tasks));
581  std::unordered_map<std::reference_wrapper<const TaskType>, std::size_t, std::hash<TaskType>, std::equal_to<TaskType>> positions;
582  std::size_t i = 0;
583  for (const TaskType& task: ordered_tasks) {
584  positions.insert(std::make_pair(std::cref(task), i++));
585  }
586  std::vector<std::reference_wrapper<const MachineType>> machines;
587  GetProblem().GetMachines(std::back_inserter(machines));
588  std::sort(
589  machines.begin(), machines.end(), [](const MachineType& m1, const MachineType& m2) { return m1.GetMachineID() < m2.GetMachineID(); });
590  for (const MachineType& machine: machines) {
591  std::vector<std::reference_wrapper<const TaskType>> machine_tasks;
592  GetProblem().GetMachineTasks(std::back_inserter(machine_tasks), machine.GetMachineID());
593  std::sort(machine_tasks.begin(), machine_tasks.end(), [&positions](const auto& t1, const auto& t2) {
594  return positions.at(t1) < positions.at(t2);
595  });
596  for (const TaskType& task: machine_tasks) {
597  solution += std::to_string(task.GetJob().GetJobID()) + " ";
598  }
